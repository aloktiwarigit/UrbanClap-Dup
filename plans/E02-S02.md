**AuthModule.kt — key detail:** Prefs file name must be `"tech_auth_session"` (matching WS-A `SessionManager`).

```kotlin
// technician-app data/auth/di/AuthModule.kt (skeleton)
package com.homeservices.technician.data.auth.di

@Module
@InstallIn(SingletonComponent::class)
public object AuthModule {
    @Provides @Singleton
    public fun provideFirebaseAuth(): FirebaseAuth = FirebaseAuth.getInstance()

    @Provides @Singleton @AuthPrefs
    public fun provideAuthPrefs(@ApplicationContext context: Context): SharedPreferences {
        val masterKeyAlias = MasterKeys.getOrCreate(MasterKeys.AES256_GCM_SPEC)
        return EncryptedSharedPreferences.create(
            "tech_auth_session",   // ← differs from customer-app "auth_session"
            masterKeyAlias, context,
            EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
            EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM,
        )
    }
}
```
